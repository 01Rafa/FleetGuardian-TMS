- [ ] **Step 4: Create `frontend/src/i18n/index.js`**

```js
import i18n from 'i18next'
import { initReactI18next } from 'react-i18next'
import LanguageDetector from 'i18next-browser-languagedetector'
import es from '../locales/es.json'
import en from '../locales/en.json'

i18n
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    resources: { es: { translation: es }, en: { translation: en } },
    fallbackLng: 'es',
    defaultNS: 'translation',
    detection: {
      order: ['localStorage'],
      lookupLocalStorage: 'i18nextLng',
      caches: ['localStorage'],
    },
    interpolation: { escapeValue: false },
  })

export default i18n
```

import { useTranslation } from 'react-i18next'
import i18n from '../i18n/index.js'

export default function Configuracion() {
  const { t } = useTranslation()
  const currentLang = i18n.language?.startsWith('en') ? 'en' : 'es'

  const btnCls = (lang) =>
    `px-5 py-2 rounded-lg text-sm font-medium border transition-colors ${
      currentLang === lang
        ? 'bg-gold text-bg-deep border-gold'
        : 'bg-surface-2 text-text-muted border-border-dim hover:text-text-primary'
    }`

  return (
    <div className="max-w-2xl space-y-6">
      <h2 className="font-serif text-2xl text-text-primary">{t('settings.title')}</h2>

      <div className="bg-surface border border-border-dim rounded-xl p-5 space-y-4">
        <h3 className="font-serif text-lg text-text-primary">{t('settings.language.title')}</h3>
        <div className="flex gap-3">
          <button className={btnCls('es')} onClick={() => i18n.changeLanguage('es')}>
            {t('settings.language.spanish')}
          </button>
          <button className={btnCls('en')} onClick={() => i18n.changeLanguage('en')}>
            {t('settings.language.english')}
          </button>
        </div>
      </div>
    </div>
  )
}

import { Link, useLocation, useNavigate } from 'react-router-dom'

export function Layout({ children }) {

  const navItems = [

    await logout()
    navigate('/login')

        <div className="px-5 py-6 border-b border-border-dim">

        </div>

          {navItems.map(({ to, label }) => (
            <Link
              key={to}
              to={to}

                pathname.startsWith(to)
                  ? 'bg-gold-dim text-gold font-medium'

              {label}

            ⚙ {t('nav.settings')}
          </Link>

          </button>
        </div>
      </aside>
